import builtins # 查看内置名称空间

print(dir(builtins))

# 全局作用域
# 内置和全局名称空间都是全局作用域

# 局部作用域
# 局部名称空间是局部作用域

# 1.局部找 2.全局找 3.内置找

x = 1

def foo():
    x = 2
